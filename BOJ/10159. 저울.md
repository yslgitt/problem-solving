N = int(input())
M = int(input())

grid = [[0 for _ in range(N+1)] for _ in range(N+1)]

    x, y = map(int,input().split())
    grid[x][y] = 1

# í”Œë¡œì´ë“œ ì›Œì…œ
for k in range(1, N+1):
    for i in range(1, N+1):
        for j in range(1, N+1):
            if grid[i][k] and grid[k][j]:
                grid[i][j] = 1

for i in range(1, N+1):
    cnt = 0
    for j in range(1, N+1):
        if not grid[i][j] and not grid[j][i]:
            cnt += 1
    print(cnt-1)
