import sys
import heapq # 3ë²ˆ ì¡°ê±´
input = sys.stdin.readline

N, M = map(int,input().split())
graph = [[] for _ in range(N+1)]
nums = [0]*(N+1)
result = []

for _ in range(M):
    A, B = map(int,input().split())
    graph[A].append(B)
    nums[B] += 1

q = []
for i in range(1, N+1):
    if not nums[i]:
        q.append(i)

while q:
    now = heapq.heappop(q) 
    result.append(now) 

    for i in graph[now]:
        nums[i] -= 1
        if not nums[i]:
            heapq.heappush(q, i)

print(*result)
