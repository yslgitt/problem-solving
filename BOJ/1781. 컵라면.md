```python
import sys
import heapq

input = sys.stdin.readline

N = int(input())
prb = [list(map(int,input().split())) for _ in range(N)]
prb.sort()

result = []
for d, r in prb:
    heapq.heappush(result, r)
    if len(result) > d:
        heapq.heappop(result)

print(sum(result))
```
