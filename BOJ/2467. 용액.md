```python
N = int(input())
liquid = list(map(int,input().split()))

l, r = 0, N-1
L, R = liquid[0], liquid[-1]
mn = 1e9*2
while l < r:
    tmp = liquid[l] + liquid[r]

    if abs(tmp) < mn:
        L = liquid[l]
        R = liquid[r]
        mn = abs(tmp)

    if tmp < 0:
        l += 1
    elif tmp > 0:
        r -= 1
    else:
        break

print(L, R)
```
