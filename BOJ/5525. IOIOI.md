import sys
input = sys.stdin.readline

N = int(input())
M = int(input())
S = input().strip('O')

i, cnt, res = 0, 0, 0
while i < M-2:
    if S[i:i+3] == 'IOI':
        cnt += 1
        i += 2
        if cnt == N:
            res += 1
            cnt -= 1
    else:
        i += 1
        cnt = 0

print(res)
