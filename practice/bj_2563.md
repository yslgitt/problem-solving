grid = [[0]*101 for _ in range(101)]

N = int(input())
for _ in range(N):
    x, y = map(int,input().split())

    for i in range(10):
        for j in range(10):
            grid[x+i][y+j] += 1

res = 0
for i in range(101):
    for j in range(101):
        if grid[i][j]:
            res += 1

print(res)
