```python
N = int(input())
pillar = [list(map(int, input().split()))for _ in range(N)]
pillar.sort()

y = []
for i in range(N):
    y.append(pillar[i][1])

sum = max(y)
c = y.index(max(y))

h = pillar[0][1]
for j in range(c):
    if h < pillar[j+1][1]:
        sum += h * (pillar[j+1][0]-pillar[j][0])
        h = pillar[j+1][1]
    else:
        sum += h * (pillar[j+1][0]-pillar[j][0])

h = pillar[-1][1]
for k in range(N-1, c, -1):
    if h < pillar[k-1][1]:
        sum += h * (pillar[k][0]-pillar[k-1][0])
        h = pillar[k-1][1]
    else:
        sum += h * (pillar[k][0]-pillar[k-1][0])

print(sum)
```
