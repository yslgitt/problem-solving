def binary(lst, start, end, key):
    global now
    if start > end:
        return 0
    mid = (start + end) //2
 
    if lst[mid] == key:
        return 1
    elif lst[mid] < key:
        if now == 1:
            return 0
        now = 1
        return binary(lst,mid+1,end,key)
    else:
        if now == 2:
            return 0
        now = 2
        return binary(lst,start,mid-1,key)
