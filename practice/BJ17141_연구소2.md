import itertools
from collections import deque

#ì…ë ¥
N, M = map(int,input().split())
lab = [list(map(int,input().split())) for _ in range(N)]

mn = 9999999999 #ìµœì†Ÿê°’
virus = [] #ë°”ì´ëŸ¬ìŠ¤ ìœ„ì¹˜ ë‹´ì„ ë¦¬ìŠ¤íŠ¸
cnt = 0 

for i in range(N):
    for j in range(N):
        if lab[i][j] == 2:
            virus.append((i,j)) #ë°”ì´ëŸ¬ìŠ¤ ìœ„ì¹˜ ë‹´ê¸°
        elif lab[i][j] == 1: 
            cnt += 1 #ë²½ ìˆ˜

#ë°”ì´ëŸ¬ìŠ¤ ìœ„ì¹˜ ì¡°í•© ë‹´ê¸°
tmp_virus = list(itertools.combinations(virus,M))

#ìœ„ì¹˜ 
for tmp in tmp_virus:
    q = deque(tmp)
    visited = [[-1]*N for _ in range(N)]

    space = N * N - cnt - M #ë°”ì´ëŸ¬ìŠ¤ë¥¼ í¼íŠ¸ë ¤ì•¼ í•  ê³³ ìˆ˜
    
    #BFS
    for x,y in q:
        visited[x][y] = 0
	
    while q:
        tx, ty = q.popleft()

        for dx, dy in ((-1,0),(0,1),(1,0),(0,-1)):
            nx = tx + dx
            ny = ty + dy
            # ë°©ë¬¸ ì•ˆí–ˆê³ , ë²”ìœ„ ì•ˆ ìˆê³ , ë²½ì´ ì•„ë‹ ê²½ìš°
            if 0 <= nx < N and 0 <= ny < N and lab[nx][ny] != 1 and visited[nx][ny] == -1:
                visited[nx][ny] = visited[tx][ty] + 1
                space -= 1 
                q.append((nx,ny))

    if space == 0: #ëª¨ë‘ í¼íŠ¸ë ¸ì„ ë•Œ
        time = visited[tx][ty]
        if mn > time:
            mn = time


if mn == 9999999999:
    print(-1)
else:
    print(mn)
